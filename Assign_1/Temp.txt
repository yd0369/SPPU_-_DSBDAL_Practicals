# What is dictionary ?
# It is key-value pair

di = { 'Roll':[2, 4, 8, 9, 10],
'Name':['Vivek', 'Aboli', 'Shrikant', 'Sita', 'Vijay'],
'Marks':['First', 'Distinction', 'Distinction', 'Second', 'First']
}

import pandas as pd

df = pd.DataFrame(di)
df

df.shape # it gives how many rows and how many columns

df.columns #This displays names of columns with their type

df.size #Total data present (rows x columns)

df.dtypes

# Above types are not necessary for our data
# Roll number float64 -> int32
# Name object -> string
# Marks -> string

df.astype({'Roll': 'int32'}).dtypes

df = df.astype({'Name': 'string'}).dtypes

df.dtypes

# Conversion of data types must be done in single statement
df = df.astype({'Name': 'string', 'Roll':'int32'}).dtypes

df

# Converting categorical data to numeric data using inplace replacing the values

df['Marks'].replace(['Distinction', 'First', 'Second'],[0, 1, 2], inplace=True)

df

df.dtypes

df["Marks"] = df["Marks"].astype('category')
df.dtypes

# This is encoding function
df["Marks"] = df["Marks"].cat.codes

df
